import React, { Component } from 'react';
import { Row, Col, FormGroup, ControlLabel, FormControl } from 'react-bootstrap';
import _ from 'lodash';
 
export default class EditProfile extends Component {
  constructor(props) {
    super(props);
    this.state = this.props.userInfo;
    this.handleValueChange = this.handleValueChange.bind(this);
  }
 
  handleValueChange(e) {
    var nextState = _.cloneDeep(this.state)
 
    nextState[e.target.name] = e.target.value
 
    this.setState(nextState);
  }
 
  render() {
    const inputs = Object.keys(this.state).filter((key) => { return key != "id" }).map((key) => {
      return (
        <FormGroup controlId={`controlFor${key}`} key={key}>
          <ControlLabel>{`${key.split("_").map((word) => { return word[0].toUpperCase() + word.substr(1, 99) }).join(" ")}:`}</ControlLabel>
          <FormControl type="text" name={key} value={this.state[key]} onChange={this.handleValueChange} />
        </FormGroup>
      );
    });
 
    return (
      <form>
        {inputs}
        <FormGroup controlId="controlForBio">
          <ControlLabel>{`Biography:`}</ControlLabel>
          <FormControl componentClass="textarea" name="bio" value={this.state.bio} onChange={this.handleValueChange} />
        </FormGroup>
      </form>
    );
  }
}
